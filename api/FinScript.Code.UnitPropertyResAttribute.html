UnitPropertyResAttribute is a general-purpose attribute to specify user-visible localizable strings for types and members. The string properties of this class can be used either as literals or as resource identifiers into a specified ResourceType

[AttributeUsage(AttributeTargets.Property)]
[MeansImplicitUse]
public class UnitPropertyResAttribute : Attribute, IUnitPropertyAttribute, IPropertyDisplayParameters
